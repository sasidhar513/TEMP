void naiveStringMatching(string t ,string p)
{
    //cout<<"im calleed"<<endl;
    int n=t.length();
    int m=p.length();
    for(int i=0;i<n;i++)
    {
        bool yes=true;
        for(int j=0;j<m;j++)
        {
            
            if(t[i+j]!=p[j])
            {
                yes=false;
                break;
            }
        }
        if(yes)
            cout<<"pattern found at "<<i<<endl;
    }
}
void improvedNaiveStringMatching(string t,string p)
{
    int n=t.length();
    int m=p.length();
    int check[m];
    for(int i=0;i<m;i++)
        check[i]=0;
    bool unique=true;
    for(int i=0;i<m;i++)
    {
        check[t[i]]++;
        if(check[t[i]]>1)
        {
            unique=false;
            break;
        }            
    }
    if(!unique)
    {
        naiveStringMatching(t,p);
        return;
    }
    int j=0,i=0;
   while(true)
   {
        
        if(t[i]==t[j])
        {
            j++;
            
        }
        if(j==m)
            cout<<"pattern found at "<< i-j<<endl;
        else
        {
            j=0;
        }
         i++;
    }
    
    
}
